# Sync Code - Complete Redesign & Technical Implementation Plan

## Project Overview

**App Name**: Sync Code  
**Purpose**: A collaborative real-time code synchronization platform for group coding sessions with video, chat, and whiteboard  
**Current State**: Functional core with basic UI; needs complete visual overhaul + full-stack enhancement  
**Tech Stack**: 
- Frontend: React with modern UI components
- Database: Neon PostgreSQL
- Authentication: NextAuth.js or Auth0 with Neon
- Backend: Next.js API routes or Node.js
- Real-time: WebSockets (Socket.io or similar)
- Deployment: Vercel (frontend) + Render/Railway (backend)

**Goal**: Production-grade redesign with happy aesthetics, dark/light themes, new features, complete auth system, and robust database architecture

## PART 1: AUTHENTICATION & DATABASE ARCHITECTURE

### 1.1 Database Schema (Neon PostgreSQL)

#### **Users Table**
```sql
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email VARCHAR(255) UNIQUE NOT NULL,
  username VARCHAR(100) UNIQUE NOT NULL,
  full_name VARCHAR(255),
  password_hash VARCHAR(255), -- For email/password auth
  avatar_url TEXT,
  bio TEXT,
  theme_preference VARCHAR(10) DEFAULT 'light', -- 'light' or 'dark'
  email_verified BOOLEAN DEFAULT false,
  email_verified_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  last_login_at TIMESTAMP,
  deleted_at TIMESTAMP -- Soft delete
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_username ON users(username);
```

#### **OAuth Providers Table** (GitHub, Google)
```sql
CREATE TABLE oauth_accounts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  provider VARCHAR(50) NOT NULL, -- 'github', 'google'
  provider_account_id VARCHAR(255) NOT NULL,
  provider_user_data JSONB, -- Store profile info
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(provider, provider_account_id)
);

CREATE INDEX idx_oauth_accounts_user_id ON oauth_accounts(user_id);
```

#### **Sessions Table** (Coding Sessions)
```sql
CREATE TABLE coding_sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  creator_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  title VARCHAR(255) NOT NULL,
  description TEXT,
  code_content TEXT DEFAULT '',
  language VARCHAR(50) DEFAULT 'javascript', -- python, java, cpp, etc.
  is_public BOOLEAN DEFAULT false,
  is_active BOOLEAN DEFAULT true,
  session_code VARCHAR(10) UNIQUE NOT NULL, -- Shareable code (e.g., "AB12CD34")
  started_at TIMESTAMP DEFAULT NOW(),
  ended_at TIMESTAMP,
  max_participants INT DEFAULT 10,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  archived_at TIMESTAMP -- Soft delete
);

CREATE INDEX idx_coding_sessions_creator_id ON coding_sessions(creator_id);
CREATE INDEX idx_coding_sessions_session_code ON coding_sessions(session_code);
CREATE INDEX idx_coding_sessions_is_active ON coding_sessions(is_active);
```

#### **Session Participants Table**
```sql
CREATE TABLE session_participants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID NOT NULL REFERENCES coding_sessions(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  joined_at TIMESTAMP DEFAULT NOW(),
  left_at TIMESTAMP,
  lines_of_code_contributed INT DEFAULT 0,
  is_host BOOLEAN DEFAULT false,
  color_indicator VARCHAR(7) DEFAULT '#FF6B5B', -- Color for cursor
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(session_id, user_id)
);

CREATE INDEX idx_session_participants_session_id ON session_participants(session_id);
CREATE INDEX idx_session_participants_user_id ON session_participants(user_id);
```

#### **Code Versions Table** (For session history)
```sql
CREATE TABLE code_versions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID NOT NULL REFERENCES coding_sessions(id) ON DELETE CASCADE,
  version_number INT NOT NULL,
  code_content TEXT NOT NULL,
  changed_by UUID NOT NULL REFERENCES users(id),
  change_summary VARCHAR(255),
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(session_id, version_number)
);

CREATE INDEX idx_code_versions_session_id ON code_versions(session_id);
```

#### **Chat Messages Table**
```sql
CREATE TABLE chat_messages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID NOT NULL REFERENCES coding_sessions(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  message_content TEXT NOT NULL,
  message_type VARCHAR(20) DEFAULT 'text', -- 'text', 'code_snippet', 'system'
  code_snippet_language VARCHAR(50), -- If type is code_snippet
  reactions JSONB DEFAULT '{}', -- {"emoji": ["user_id1", "user_id2"]}
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  deleted_at TIMESTAMP -- Soft delete
);

CREATE INDEX idx_chat_messages_session_id ON chat_messages(session_id);
CREATE INDEX idx_chat_messages_user_id ON chat_messages(user_id);
```

#### **Whiteboard Data Table**
```sql
CREATE TABLE whiteboard_data (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID NOT NULL REFERENCES coding_sessions(id) ON DELETE CASCADE,
  stroke_data JSONB NOT NULL, -- SVG paths, shapes data
  created_by UUID NOT NULL REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW(),

```

#### **User Activity Log Table** (For metrics & analytics)
```sql
CREATE TABLE activity_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,

  activity_type VARCHAR(50) NOT NULL, -- 'session_created', 'session_joined', 'code_lines_added', etc.
  activity_data JSONB, -- Extra context

CREATE INDEX idx_activity_log_user_id ON activity_log(user_id);

```

#### **Achievements/Badges Table**
```sql
CREATE TABLE achievements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
```

#### **User Statistics Table** (Cached metrics)
```sql
CREATE TABLE user_statistics (
```

### 1.2 Authentication System (NextAuth.js + Neon)

#### **Authentication Methods Supported**
1. **Email + Password** (Traditional)
2. **GitHub OAuth**
3. **Google OAuth**
4. **Magic Link** (Passwordless email)

#### **NextAuth Configuration Example**
```javascript
// pages/api/auth/[...nextauth].js
import NextAuth from "next-auth"
import CredentialsProvider from "next-auth/providers/credentials"
import GitHubProvider from "next-auth/providers/github"
import GoogleProvider from "next-auth/providers/google"
import { neon } from "@neondatabase/serverless"

const sql = neon(process.env.DATABASE_URL)

export default NextAuth({
  providers: [
    // Email/Password Provider
    CredentialsProvider({
      name: 'Credentials',
      credentials: {
        email: { label: "Email", type: "email" },
        password: { label: "Password", type: "password" }
      },
      async authorize(credentials) {
        // Hash & verify password from Neon DB
        const user = await sql`
          SELECT id, email, username, full_name, avatar_url 
          FROM users 
          WHERE email = ${credentials.email}
        `
        // Verify password hash...
        return user[0] || null
      }
    }),
    
    // GitHub OAuth
    GitHubProvider({
      clientId: process.env.GITHUB_ID,
      clientSecret: process.env.GITHUB_SECRET,
      async profile(profile) {
        // Insert/update in oauth_accounts table
        return {
          id: profile.id,
          name: profile.name,
          email: profile.email,
          image: profile.avatar_url
        }
      }
    }),
    
    // Google OAuth
    GoogleProvider({
      clientId: process.env.GOOGLE_ID,
      clientSecret: process.env.GOOGLE_SECRET,
      async profile(profile) {
        return {
          id: profile.id,
          name: profile.name,
          email: profile.email,
          image: profile.picture
        }
      }
    })
  ],
  
  database: process.env.DATABASE_URL,
  
  callbacks: {
    async jwt({ token, user, account }) {
      if (user) {
        token.id = user.id
        token.username = user.username
      }
      return token
    },
    async session({ session, token }) {
      session.user.id = token.id
      session.user.username = token.username
      return session
    }
  },
  
  pages: {
    signIn: '/auth/signin',
    signUp: '/auth/signup',
    error: '/auth/error',
    verifyRequest: '/auth/verify-email'
  },
  
  session: {
    strategy: 'jwt',
    maxAge: 30 * 24 * 60 * 60, // 30 days
    updateAge: 24 * 60 * 60 // Update session every 24h
  }
})
```

#### **Session Management**
- JWT-based sessions stored client-side (secure, httpOnly cookies)
- Session refresh on user activity
- CSRF protection enabled
- Secure cookies in production (https only)

### 1.4 Real-time Communication

#### **WebSocket Events** (Using Socket.io)
```javascript
// Server-side socket handlers

// Code synchronization
socket.on('code:update', (data) => {
  // Broadcast code change to all participants
  io.to(sessionId).emit('code:update', data)
  // Save to Neon DB
  await sql`UPDATE coding_sessions SET code_content = $1 WHERE id = $2`
})

// Chat
socket.on('chat:send', (message) => {
  io.to(sessionId).emit('chat:message', message)
  await sql`INSERT INTO chat_messages (...) VALUES (...)`
})

// Whiteboard
socket.on('whiteboard:stroke', (stroke) => {
  io.to(sessionId).emit('whiteboard:update', stroke)
})

// Cursor tracking
socket.on('cursor:move', (position) => {
  socket.broadcast.to(sessionId).emit('cursor:position', {
    userId,
    position
  })
})

// User presence
socket.on('user:typing', (data) => {
  socket.broadcast.to(sessionId).emit('user:typing', data)
})

socket.on('user:status', (status) => {
  io.to(sessionId).emit('user:status', { userId, status })
})
```

## PART 6: DATABASE QUERIES & OPERATIONS

### Common Queries

#### **Get User with Stats**
```sql
SELECT 
  u.*, 
  s.total_sessions_created,
  s.total_collaborators,
  s.total_lines_of_code
FROM users u
LEFT JOIN user_statistics s ON u.id = s.user_id
WHERE u.id = $1
```

#### **Get Active Sessions for User**
```sql
SELECT cs.* FROM coding_sessions cs
WHERE cs.is_active = true 
  AND (cs.creator_id = $1 OR cs.id IN (
    SELECT session_id FROM session_participants WHERE user_id = $1
  ))
ORDER BY cs.updated_at DESC
```

#### **Get Session Details with Participants**
```sql
SELECT 
  cs.*,
  json_agg(json_build_object(
    'user_id', sp.user_id,
    'username', u.username,
    'avatar_url', u.avatar_url,
    'is_host', sp.is_host,
    'joined_at', sp.joined_at
  )) as participants
FROM coding_sessions cs
LEFT JOIN session_participants sp ON cs.id = sp.session_id
LEFT JOIN users u ON sp.user_id = u.id
WHERE cs.id = $1
GROUP BY cs.id
```

#### **Get Recent Chat for Session**
```sql
SELECT cm.*, u.username, u.avatar_url
FROM chat_messages cm
JOIN users u ON cm.user_id = u.id
WHERE cm.session_id = $1
ORDER BY cm.created_at DESC
LIMIT 50
```

#### **Calculate User Statistics** (Run periodically)
```sql
UPDATE user_statistics SET
  total_sessions_created = (
    SELECT COUNT(*) FROM coding_sessions WHERE creator_id = $1
  ),
  total_sessions_joined = (
    SELECT COUNT(*) FROM session_participants WHERE user_id = $1
  ),
  total_collaborators = (
    SELECT COUNT(DISTINCT sp.user_id) 
    FROM session_participants sp
    JOIN session_participants sp2 ON sp.session_id = sp2.session_id
    WHERE sp2.user_id = $1 AND sp.user_id != $1
  ),
  total_lines_of_code = (
    SELECT SUM(lines_of_code_contributed) 
    FROM session_participants WHERE user_id = $1
  ),
  updated_at = NOW()
WHERE user_id = $1
```

#### **Unlock Achievement for User**
```sql
INSERT INTO user_achievements (user_id, achievement_id, unlocked_at)
SELECT $1, a.id, NOW()
FROM achievements a
WHERE a.badge_name = $2
  AND NOT EXISTS (
    SELECT 1 FROM user_achievements 
    WHERE user_id = $1 AND achievement_id = a.id
  )
RETURNING *
```
